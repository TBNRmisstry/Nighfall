## ⚡ Update `index.js` for Slash Commands

```
JavaScript

require('dotenv').config();
const fs = require('fs');
const { Client, GatewayIntentBits, Collection } = require('discord.js');

const client = new Client({
  intents: [GatewayIntentBits.Guilds],
});

client.commands = new Collection();

// Load commands
const commandFiles = fs.readdirSync('./commands').filter(file => file.endsWith('.js'));

for (const file of commandFiles) {
  const command = require(`./commands/${file}`);
  client.commands.set(command.data.name, command);
}

client.once('ready', () => {
  console.log(`Logged in as ${client.user.tag}`);
});

// Handle interactions
client.on('interactionCreate', async interaction => {
  if (!interaction.isChatInputCommand()) return;

  const command = client.commands.get(interaction.commandName);
  if (!command) return;

  try {
    await command.execute(interaction);
  } catch (error) {
    console.error(error);
    await interaction.reply({ content: 'Error executing command!', ephemeral: true });
  }
});

client.login(process.env.TOKEN);
```

const path = require('path');

function loadCommands(dir) {
  const files = fs.readdirSync(dir, { withFileTypes: true });

  for (const file of files) {
    if (file.isDirectory()) {
      loadCommands(path.join(dir, file.name));
    } else if (file.name.endsWith('.js')) {
      const command = require(path.join(dir, file.name));
      client.commands.set(command.data.name, command);
    }
  }
}

# 🎭 3\. Queue System (Core Gameplay Logic)

### `systems/queueSystem.js`

```
JavaScript

const queue = [];

function joinQueue(user) {
  if (queue.includes(user.id)) return false;
  queue.push(user.id);
  return true;
}

function leaveQueue(user) {
  const index = queue.indexOf(user.id);
  if (index === -1) return false;
  queue.splice(index, 1);
  return true;
}

function getQueue() {
  return queue;
}

function startMatch() {
  if (queue.length < 2) return null;

  const players = [...queue];
  queue.length = 0;

  const killer = players[Math.floor(Math.random() * players.length)];
  const survivors = players.filter(p => p !== killer);

  return { killer, survivors };
}

module.exports = { joinQueue, leaveQueue, getQueue, startMatch };
```

require('dotenv').config();
const { Client, GatewayIntentBits, Collection } = require('discord.js');

const client = new Client({
  intents: [GatewayIntentBits.Guilds],
});

client.commands = new Collection();

// Load command
const setupCommand = require('./commands/setup');
client.commands.set(setupCommand.data.name, setupCommand);

client.once('ready', () => {
  console.log(`✅ Logged in as ${client.user.tag}`);
});

client.on('interactionCreate', async interaction => {
  if (!interaction.isChatInputCommand()) return;

  const command = client.commands.get(interaction.commandName);
  if (!command) return;

  try {
    await command.execute(interaction);
  } catch (err) {
    console.error(err);
    interaction.reply({ content: 'Error!', ephemeral: true });
  }
});

client.login(process.env.TOKEN);
```

# 🧱 `commands/setup.js` (FULL Forsaken System)

```
JavaScript

const {
  SlashCommandBuilder,
  PermissionFlagsBits,
  ChannelType,
} = require('discord.js');

module.exports = {
  data: new SlashCommandBuilder()
    .setName('setup')
    .setDescription('Build Forsaken-style server')
    .setDefaultMemberPermissions(PermissionFlagsBits.Administrator),

  async execute(interaction) {
    const guild = interaction.guild;

    await interaction.reply('⚙️ Building Forsaken server...');

    // ===== ROLES =====
    const verified = await guild.roles.create({ name: 'Verified' });
    const dev = await guild.roles.create({ name: 'Developer' });
    const staff = await guild.roles.create({ name: 'Staff' });

    // ===== LOCK PERMS =====
    const lockEveryone = [\
      {\
        id: guild.roles.everyone.id,\
        deny: ['ViewChannel'],\
      },\
    ];

    const readOnly = [\
      {\
        id: guild.roles.everyone.id,\
        allow: ['ViewChannel'],\
        deny: ['SendMessages'],\
      },\
    ];

    // ===== TOP PRIVATE CHANNELS =====
    await guild.channels.create({
      name: 'dev zone',
      type: ChannelType.GuildText,
      permissionOverwrites: [\
        { id: guild.roles.everyone.id, deny: ['ViewChannel'] },\
        { id: dev.id, allow: ['ViewChannel'] },\
      ],
    });

    await guild.channels.create({
      name: 'game zone',
      type: ChannelType.GuildText,
      permissionOverwrites: [\
        { id: guild.roles.everyone.id, deny: ['ViewChannel'] },\
        { id: staff.id, allow: ['ViewChannel'] },\
      ],
    });

    // ===== COMMUNITY =====
    const community = await guild.channels.create({
      name: 'Community',
      type: ChannelType.GuildCategory,
    });

    const communityChannels = [\
      'videos',\
      'fan-art',\
      'events',\
      'staff-shitpost',\
      'poll-responses',\
    ];

    for (const ch of communityChannels) {
      await guild.channels.create({
        name: ch,
        type: ChannelType.GuildText,
        parent: community.id,
        permissionOverwrites: readOnly,
      });
    }

    // ===== CHATS =====
    const chats = await guild.channels.create({
      name: 'Chats',
      type: ChannelType.GuildCategory,
    });

    const chatChannels = ['general', 'bot-commands', 'forsaken', 'creations'];

    for (const ch of chatChannels) {
      await guild.channels.create({
        name: ch,
        type: ChannelType.GuildText,
        parent: chats.id,
      });
    }

    // ===== VOICE =====
    const voice = await guild.channels.create({
      name: 'Voice Chats',
      type: ChannelType.GuildCategory,
    });

    await guild.channels.create({
      name: 'General 1',
      type: ChannelType.GuildVoice,
      parent: voice.id,
    });

    // ===== INFORMATION =====
    const info = await guild.channels.create({
      name: 'Information',
      type: ChannelType.GuildCategory,
    });

    const infoChannels = [\
      'rules',\
      'how-to-report-or-appeal',\
      'faq',\
      'ping-roles',\
      'links',\
      'access-roles',\
    ];

    for (const ch of infoChannels) {
      await guild.channels.create({
        name: ch,
        type: ChannelType.GuildText,
        parent: info.id,
        permissionOverwrites: readOnly,
      });
    }

    // ===== IMPORTANT =====
    const important = await guild.channels.create({
      name: 'Important',
      type: ChannelType.GuildCategory,
    });

    const importantChannels = [\
      'announcements',\
      'sub-announcements',\
      'update-log',\
      'dev-log',\
      'polls',\
      'ugc',\
      'socials',\
    ];

    for (const ch of importantChannels) {
      await guild.channels.create({
        name: ch,
        type: ChannelType.GuildAnnouncement,
        parent: important.id,
      });
    }

    await interaction.editReply('✅ Forsaken-style server created!');
  },
};
